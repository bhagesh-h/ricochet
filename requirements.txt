# ── System / Runtime Requirements ──────────────────────────
#
# Ricochet orchestrates Docker containers to execute bioinformatics
# tools.  Docker must be installed and running BEFORE launching
# the app.
#
# ┌──────────────┬───────────────────────────────────────────────────────────┐
# │ OS           │ Docker installation                                        │
# ├──────────────┼───────────────────────────────────────────────────────────┤
# │ macOS        │ Docker Desktop ≥ 4.x                                       │
# │              │ https://docs.docker.com/desktop/install/mac-install/       │
# │              │ Apple Silicon: enable Rosetta 2 in Docker Desktop settings │
# ├──────────────┼───────────────────────────────────────────────────────────┤
# │ Windows 10+  │ Docker Desktop ≥ 4.x with WSL2 backend (recommended)      │
# │              │ https://docs.docker.com/desktop/install/windows-install/   │
# │              │ Enable WSL2 in Windows Features before installing          │
# ├──────────────┼───────────────────────────────────────────────────────────┤
# │ Linux        │ Option A — Docker Desktop for Linux (GUI):                 │
# │              │   https://docs.docker.com/desktop/install/linux-install/   │
# │              │                                                             │
# │              │ Option B — Docker Engine (CLI only, lighter):              │
# │              │   Ubuntu/Debian : sudo apt-get install docker.io           │
# │              │   Fedora/RHEL   : sudo dnf install docker-ce               │
# │              │   Arch Linux    : sudo pacman -S docker                    │
# │              │                                                             │
# │              │ Required post-install step (both options):                 │
# │              │   sudo usermod -aG docker $USER                            │
# │              │   (log out and back in for the change to take effect)      │
# │              │                                                             │
# │              │ To start automatically on boot:                            │
# │              │   sudo systemctl enable --now docker                       │
# └──────────────┴───────────────────────────────────────────────────────────┘
#
# Docker version: 24.x or newer recommended.
# docker>=24.0.0  (system package, not a pip/pub package)
